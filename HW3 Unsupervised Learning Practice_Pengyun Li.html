#Now let's plot the Elbow Curve to look for proper k.

from sklearn.cluster import KMeans
from matplotlib import pyplot as plt

within_cluster_variance = [] # initiate empty list with no data

for k in range(2, 20): 
    kmeans = KMeans(n_clusters=k)
    kmeans.fit(scaled_red)
    within_cluster_variance.append(kmeans.inertia_)

fig = plt.figure(figsize=(15, 5))
plt.plot(range(2, 20), within_cluster_variance, marker='o')
plt.grid(True)
plt.xlabel('Number of clusters')
plt.ylabel('Reduction in within cluster variation')
plt.title('Elbow Curve');

We cannot see a very clear elbow from the curve above, so I will test for k = 5,6,7 as they appear to be possible elbow points.
